CREATE SCHEMA PUBLIC AUTHORIZATION DBA
CREATE MEMORY TABLE USERS(USER_ VARCHAR(50) NOT NULL PRIMARY KEY,PASSWORD VARCHAR(50),NAME VARCHAR(50),SURNAME VARCHAR(50),EMAIL VARCHAR(50),IDPROFILE INTEGER)
CREATE MEMORY TABLE AUTONUMERICAL(STATE INTEGER,AUTONUMERICAL INTEGER)
CREATE MEMORY TABLE PROFILES(IDPROFILE INTEGER NOT NULL PRIMARY KEY,PROFILE VARCHAR(50),XMLCLIENT LONGVARCHAR,XMLSERVER LONGVARCHAR)
ALTER TABLE USERS ADD CONSTRAINT SYS_FK_95 FOREIGN KEY(IDPROFILE) REFERENCES PROFILES(IDPROFILE)
CREATE VIEW VUSERS (USER_,PASSWORD,NAME,SURNAME,EMAIL,IDPROFILE,PROFILE,XMLCLIENT,XMLSERVER) AS SELECT USERS.USER_,  USERS.PASSWORD,  USERS.NAME,  USERS.SURNAME,  USERS.EMAIL,  USERS.IDPROFILE, PROFILES.PROFILE,  PROFILES.XMLCLIENT, PROFILES.XMLSERVER FROM USERS LEFT JOIN PROFILES ON  PROFILES.IDPROFILE =  USERS.IDPROFILE
CREATE USER SA PASSWORD ""
CREATE USER QUICKSTART PASSWORD "QUICKSTART"
GRANT DBA TO SA
GRANT DBA TO QUICKSTART
SET WRITE_DELAY 10
SET SCHEMA PUBLIC
